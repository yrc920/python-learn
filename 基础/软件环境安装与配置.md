### 3.  脚本验证

<span style='color:blue;'>GPU版本的Pytorch 1.5.0安装成功之后，可执行如下脚本，验证GPU是否生效</span>：

```python

import torch

# 打印PyTorch版本
print("Pytorch版本：{}".format(torch.__version__))

# 检查GPU是否可用
if torch.cuda.is_available():
    print("CUDA版本：{}".format(torch.version.cuda))
    print("GPU型号：{}".format(torch.cuda.get_device_name(0)))

    import pynvml  # 需提前执行pip install pynvml

    pynvml.nvmlInit()  # 初始化NVML

    device_count = pynvml.nvmlDeviceGetCount()          # 获取GPU数量
    print(f"可用GPU数量: {device_count}")
    handle = pynvml.nvmlDeviceGetHandleByIndex(0)       # 0表示显卡标号
    meminfo = pynvml.nvmlDeviceGetMemoryInfo(handle)    # 在每一个要查看的地方都要重新定义一个meminfo
    print("总显存大小：", meminfo.total / 1024 ** 3)     # 总的显存大小，单位为GB
    print("已用显存大小：", meminfo.used / 1024 ** 3)    # 已用显存大小，单位为GB
    print("剩余显存大小：", meminfo.free / 1024 ** 3)    # 剩余显存大小，单位为GB
    pynvml.nvmlShutdown()                              # 关闭NVML
else:
    print("CUDA不可用，Pytorch当前配置可能没有安装GPU支持")

print("GPU数量：{}".format(torch.cuda.device_count()))

# 使用GPU进行计算
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
x = torch.ones(5, 5).to(device)
y = torch.randn(5, 5).to(device)
z = x + y
print(z)

```
